## Example Usage

``` r
library(harmonious)

seed <- abs(.Random.seed[4])
set.seed(seed)
coef_hyper=5
sd_hyper=.1
rhat_threshold=1.06
maxCycles=3
saveDir <- getwd()

aux_envir <- genData(
  P=250,
  I=75,
  J=3,
  K=3,
  isCorrI=FALSE
)

model <- CreateMod(
  aux_envir=aux_envir,
  coef_hyper=coef_hyper,
  sd_hyper=sd_hyper,
  nWarmup_init=2000,
  nSamples_init=2000,
  nWarmup_run=2000,
  nSamples_run=2000
)

fileDetails <- paste0(seed, "_P", P, "_I", I, "_J", J, "_K", K)

model$initialize()
model$sample()
rhatCheck()

genReport(saveDir=saveDir, fileDetails=fileDetails)
```
